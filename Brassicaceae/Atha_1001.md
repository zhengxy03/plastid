# minor frequency
```
mkdir -p minor_freq
echo "总VCF文件数:" $(ls vcf/*.vcf.gz 2>/dev/null | wc -l)
ls vcf/*.vcf.gz | head -5

# 重新计算minor frequency
for vcf_file in vcf/*.vcf.gz; do
    sample=$(basename "$vcf_file" .vcf.gz)
    output_file="minor_freq/${sample}_minor_freq.tsv"
    
    echo "处理: $sample"
    
    # 使用正确的查询格式
    bcftools query -f '%CHROM\t%POS\t%REF\t%ALT\t[%AD]\t[%AF]\t%DP\t%QUAL\n' "$vcf_file" 2>/dev/null | \
        awk -F'\t' 'BEGIN{
            OFS="\t"
            print "Chr\tPos\tRef\tAlt\tRefDepth\tAltDepth\tTotalDepth\tRefFreq\tAltFreq\tMinorFreq\tMajorAllele\tMinorAllele\tAF\tQUAL"
        } 
        {
            # 解析AD字段（格式如：974,15）
            split($5, ad, ",")
            ref_depth = ad[1] + 0
            alt_depth = ad[2] + 0
            total_depth = ref_depth + alt_depth
            
            if (total_depth > 0) {
                ref_freq = ref_depth / total_depth
                alt_freq = alt_depth / total_depth
                
                # 解析AF字段（格式如：0.015）
                af = $6 + 0
                
                # 计算minor frequency（取两个频率中较小的）
                if (alt_freq > ref_freq) {
                    minor_freq = ref_freq
                    major_allele = $4
                    minor_allele = $3
                } else {
                    minor_freq = alt_freq
                    major_allele = $3
                    minor_allele = $4
                }
                
                # 输出
                print $1, $2, $3, $4, ref_depth, alt_depth, total_depth, 
                      sprintf("%.4f", ref_freq), sprintf("%.4f", alt_freq), 
                      sprintf("%.4f", minor_freq), major_allele, minor_allele,
                      af, $8
            }
        }' > "$output_file"
    
    line_count=$(tail -n +2 "$output_file" | wc -l)
    echo "  生成: $output_file ($line_count 行)"
    
    # 显示前几行示例
    if [ "$line_count" -gt 0 ]; then
        echo "  示例:"
        head -3 "$output_file" | tail -2 | while read line; do
            echo "    $line"
        done
    fi
    echo ""
done

for freq_file in minor_freq/*_minor_freq.tsv; do
    sample=$(basename "$freq_file" _minor_freq.tsv)
    output_file="${sample}_filtered.tsv"
    
    echo "筛选样本: $sample"
    
    # 筛选minor frequency > 0.01的位点
    awk -F'\t' 'BEGIN{OFS="\t"} 
        NR==1 {print $0; next}  # 保留表头

    total_count=$(tail -n +2 "$freq_file" | wc -l)
    filtered_count=$(tail -n +2 "$output_file" | wc -l)
    
    echo "  总位点数: $total_count"
    echo "  筛选后(minorfreq>0.01): $filtered_count"
    
    # 显示前几个筛选结果
    if [ "$filtered_count" -gt 0 ]; then
        echo "  前3个筛选结果:"
        head -4 "$output_file" | tail -3 | while read line; do
            echo "    $line"
        done
    fi
    echo ""
done
